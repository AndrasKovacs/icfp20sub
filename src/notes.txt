-- Church natural numbers
--------------------------------------------------------------------------------
let Nat : U
 = (n : U) → (n → n) → n → n in

let zero : Nat
 = λ n s z. z in

let suc : Nat → Nat
 = λ a n s z. s (a n s z) in

let Vec : U → Nat → U
 = λ a n. (V : Nat → U) → V zero → ({n} → a → V n → V (suc n)) → V n in

let vnil : {a} → Vec a zero
 = λ V n c. n in

let vcons : {a n} → a → Vec a n → Vec a (suc n)
 = λ a as V n c. c a (as V n c) in
